# Question 20:
# Dictionary me sirf even values ko filter karo

# dictionary create karna
d = {"a": 1, "b": 2, "c": 3, "d": 4, "e": 5}

# empty dictionary banana
even = {}

# even values filter karna
for k, v in d.items():
    if v % 2 == 0:
        even[k] = v

# output print karna
print(even)
📊 Output:

{'b': 2, 'd': 4}
