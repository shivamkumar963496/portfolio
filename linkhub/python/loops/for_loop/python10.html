# Question 10:
# Write a Python program to take a number as input
# and find its factorial.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# factorial store karne ke liye variable
fact = 1

# loop 1 se n tak chalega aur multiplication karega
for i in range(1, n+1):
    fact *= i

# final factorial print kiya
print("Factorial =", fact)
