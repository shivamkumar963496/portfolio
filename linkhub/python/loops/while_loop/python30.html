# Question 30:
# Simple menu-driven calculator.

# infinite loop
while True:
    print("\n1. Add")
    print("2. Subtract")
    print("3. Exit")
    
    choice = int(input("Enter choice: "))
    
    if choice == 3:
        print("Program ended")
        break
    
    # input numbers
    a = int(input("Enter first number: "))
    b = int(input("Enter second number: "))
    
    # operations
    if choice == 1:
        print("Result =", a + b)
        
    elif choice == 2:
        print("Result =", a - b)
        
    else:
        print("Invalid choice")
