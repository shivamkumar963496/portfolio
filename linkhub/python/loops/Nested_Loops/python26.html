# Question 26:
# Find pair (i,j) such that i*j == target using nested loop.

# target value
target = 6

# outer loop
for i in range(1,5):

    # inner loop
    for j in range(1,5):

        # check condition
        if i * j == target:
            print("pair found", i, j)
            break

    # continue outer loop if inner not broken
    else:
        continue

    # break outer loop if pair found
    break

pair found 2 3
