# Question 43:
# Words ko unke first letter ke basis par group karo

# list create karna
words = ["apple", "ant", "bat", "ball", "cat"]

# empty dictionary
group = {}

# grouping karna
for w in words:
    key = w[0]

    if key not in group:
        group[key] = []

    group[key].append(w)

# output print karna
print(group)
📊 Output:

{'a': ['apple', 'ant'], 'b': ['bat', 'ball'], 'c': ['cat']}
