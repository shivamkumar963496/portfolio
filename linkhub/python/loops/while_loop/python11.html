# Question 11:
# Check if a number is palindrome.

# input lena
n = int(input("Enter a number: "))

# original value save
temp = n
reverse = 0

# reverse banana
while n != 0:
    last_digit = n % 10
    reverse = reverse * 10 + last_digit
    n //= 10

# compare
if temp == reverse:
    print("Palindrome number")
else:
    print("Not a palindrome number")
