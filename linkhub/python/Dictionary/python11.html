# Question 11:
# Ek Python program likho jo string ke characters ki frequency count kare.

# string create karna
s = "banana"

# empty dictionary banana
freq = {}

# frequency count karna
for ch in s:
    freq[ch] = freq.get(ch, 0) + 1

# output print karna
print(freq)

{'b': 1, 'a': 3, 'n': 2}
