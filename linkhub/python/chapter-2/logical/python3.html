# Salary aur experience input lo aur check karo job ke liye eligible hai ya nahi

# user se salary input le rahe hain
salary = int(input("Enter salary: "))

# user se experience input le rahe hain
experience = int(input("Enter experience: "))

# condition check kar rahe hain
if salary > 30000 and experience > 2:
    print("ELIGIBLE FOR JOB")
else:
    print("NOT ELIGIBLE FOR JOB")
