# Question 9:
# Take electricity units from the user and calculate the bill based on given rates

# Taking units input
units = int(input("ENTER YOUR UNITS = "))

# Calculating electricity bill
if units <= 100:
    bill = units * 5
elif units <= 200:
    bill = units * 7
else:
    bill = units * 10

# Printing bill
print("ELECTRICITY BILL =", bill, "RS")
