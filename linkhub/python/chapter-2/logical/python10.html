# Question 10:
# Do numbers input lo aur check karo koi ek negative hai ya dono positive

# user se first number input le rahe hain
a = int(input("ENTER 1ST NUMBER = "))

# user se second number input le rahe hain
b = int(input("ENTER 2ND NUMBER = "))

# condition check kar rahe hain
if a < 0 or b < 0:
    print("At least one number is negative")
else:
    print("Both numbers are positive")
