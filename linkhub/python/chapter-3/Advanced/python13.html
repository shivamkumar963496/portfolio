# Question 13:
# Display time of day based on hour
# <12 → Morning, 12–15 → Afternoon, 16–19 → Evening, >=20 → Night

# Taking time input
time = int(input("ENTER TIME (0-23) = "))

# Time classification
if time < 12:
    print("MORNING")
elif time < 16:
    print("AFTERNOON")
elif time < 20:
    print("EVENING")
else:
    print("NIGHT")
