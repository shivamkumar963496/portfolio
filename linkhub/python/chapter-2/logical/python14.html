# Salary aur department input lo aur check karo eligible hai ya nahi

# user se salary input le rahe hain
salary = int(input("Enter salary: "))

# user se department input le rahe hain
department = input("Enter department: ")

# condition check kar rahe hain
if salary > 50000 or department == "it":
    print("ELIGIBLE")
else:
    print("NOT ELIGIBLE")
