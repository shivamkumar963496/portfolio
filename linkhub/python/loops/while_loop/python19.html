# Question 19:
# Convert decimal number to binary.

# input lena
n = int(input("Enter a decimal number: "))

# original value save
temp = n
binary = ""   # empty string

# conversion process
while n > 0:
    last_digit = n % 2
    binary = str(last_digit) + binary
    n //= 2

# result print
print(f"{temp} in binary = {binary}")
📊 Output:

Enter a decimal number: 5
5 in binary = 101
