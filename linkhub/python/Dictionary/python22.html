# Question 22:
# Ek Python program likho jo string ka first non-repeating character ka index find kare.

# string create karna
s = "leetcode"

# frequency dictionary banana
freq = {}

for ch in s:
    freq[ch] = freq.get(ch, 0) + 1

# first non-repeating character ka index nikalna
for i, ch in enumerate(s):
    if freq[ch] == 1:
        print(i)
        break
