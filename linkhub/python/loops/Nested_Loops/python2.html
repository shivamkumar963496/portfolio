# Question 2:
# Print multiplication tables from 1 to 10 using nested for loops.

# outer loop for numbers (1 to 10)
for i in range(1,11):

    # inner loop for multiplication (1 to 10)
    for j in range(1,11):
        print(i*j, end=" ")
        
    # move to next line
    print()
