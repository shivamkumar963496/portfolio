# Question 42:
# Nested dictionary ko flatten karke ek single dictionary banao

# dictionary create karna
d = {
    "a": {"x": 1, "y": 2},
    "b": {"z": 3}
}

# empty dictionary
flat = {}

# flatten karna
for k1, v1 in d.items():
    for k2, v2 in v1.items():
        new_key = k1 + "-" + k2
        flat[new_key] = v2

# output print karna
print(flat)
📊 Output:

{'a-x': 1, 'a-y': 2, 'b-z': 3}
