# Question 11:
# Determine insurance premium category based on age
# Age < 25 → High, 25–50 → Normal, >50 → Low

# Taking age input
age = int(input("ENTER AGE = "))

# Checking premium category
if age < 25:
    print("HIGH PREMIUM")
elif age <= 50:
    print("NORMAL PREMIUM")
else:
    print("LOW PREMIUM")
