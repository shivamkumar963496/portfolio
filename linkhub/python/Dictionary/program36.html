# Question 36:
# Do dictionaries ki common keys nikaalo

# dictionaries create karna
d1 = {"a": 1, "b": 2, "c": 3}
d2 = {"b": 5, "c": 6, "d": 7}

# keys ko set me convert karna
keys1 = set(d1.keys())
keys2 = set(d2.keys())

# common keys nikalna
common = keys1 & keys2

# output print karna
print(common)
📊 Output:

{'b', 'c'}
