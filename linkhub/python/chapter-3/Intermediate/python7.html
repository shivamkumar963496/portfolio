# Question 7:
# Take a string from the user and classify it as SMALL, MEDIUM, or LARGE based on length

# Taking string input
s = input("ENTER STRING = ")

# Finding length of string
length = len(s)

# Checking string size
if length < 5:
    print("SMALL STRING")
elif 5 <= length <= 10:
    print("MEDIUM STRING")
else:
    print("LARGE STRING")
