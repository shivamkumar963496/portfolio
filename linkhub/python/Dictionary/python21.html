# Question 21:
# Ek Python program likho jo anagrams ko group kare.

# words list create karna
words = ["eat", "tea", "tan", "ate", "nat", "bat"]

# empty dictionary banana
d = {}

# anagram grouping karna
for w in words:
    key = "".join(sorted(w))
    d.setdefault(key, []).append(w)

# output print karna
print(list(d.values()))
📊 Output:

[['eat', 'tea', 'ate'], ['tan', 'nat'], ['bat']]
