# Marks input lo aur check karo good score range me hai ya nahi (75 se 89)

# user se marks input le rahe hain
marks = int(input("Enter marks: "))

# good marks range condition check kar rahe hain
if marks >= 75 and marks < 90:
    print("Good Score")
else:
    print("Not in range")
