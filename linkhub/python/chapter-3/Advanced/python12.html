# Question 12:
# Check whether both numbers are even, both odd, or mixed

# Taking first number input
a = int(input("ENTER FIRST NUMBER = "))

# Taking second number input
b = int(input("ENTER SECOND NUMBER = "))

# Checking conditions
if a % 2 == 0 and b % 2 == 0:
    print("BOTH EVEN")
elif a % 2 != 0 and b % 2 != 0:
    print("BOTH ODD")
else:
    print("MIXED")
