# Ek number input lo aur check karo 3 ya 5 me se kisi ek se divisible hai ya nahi

# user se number input le rahe hain
num = int(input("Enter number: "))

# condition check kar rahe hain
if num % 3 == 0 or num % 5 == 0:
    print("Divisible by 3 or 5")
else:
    print("Not divisible")
