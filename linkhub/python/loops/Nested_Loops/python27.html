# Question 27:
# Print diamond pattern but skip middle row using continue.

# total rows
n = 5

# upper part
for i in range(1, n+1):

    # skip middle row
    if i == (n // 2) + 1:
        continue

    print("  " * (n - i), end=" ")
    print("* " * (2 * i - 1))

# lower part
for i in range(n-1, 0, -1):

    # skip middle row
    if i == (n // 2) + 1:
        continue

    print("  " * (n - i), end=" ")
    print("* " * (2 * i - 1))

*
* * *
* * * * * * *
* * *
*
