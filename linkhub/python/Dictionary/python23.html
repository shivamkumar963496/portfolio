# Question 23:
# Ek Python program likho jo do dictionaries ko merge karke unki values ka sum kare.

# dictionaries create karna
d1 = {"a": 1, "b": 2}
d2 = {"b": 3, "c": 4}

# new dictionary banana
d = {}

# merge + sum logic
for k in set(d1) | set(d2):
    d[k] = d1.get(k, 0) + d2.get(k, 0)

# output print karna
print(d1)
print(d2)
print(d)
📊 Output:

{'a': 1, 'b': 2}
{'b': 3, 'c': 4}
{'a': 1, 'b': 5, 'c': 4}
