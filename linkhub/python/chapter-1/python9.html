# String number ko integer me convert karo aur conversion se pehle aur baad ka data type print karo

# num me string value store ki gayi hai
num = "100"

# conversion se pehle data type check kar rahe hain
print(type(num))

# string ko integer me convert kar rahe hain

# conversion ke baad data type check kar rahe hain
print(type(num))

# ab num integer hai isliye 50 add kar rahe hain
print(num + 50)
