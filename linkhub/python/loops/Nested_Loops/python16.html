# Question 16:
# Search a character in 2D list using nested loop and break.

# 2D list (grid)
grid = [
    ['a','b','c'],
    ['d','e','f'],
    ['g','h','i']
]

# user input
target = input("ENTER TARGET STRING LIKE A TO I : ")

# flag variable
found = False

# outer loop
for i in grid:

    # inner loop
    for j in i:

        # check match
        if j == target:
            print("found :", j)
            found = True
            break

    # break outer loop
    if found:
        break
