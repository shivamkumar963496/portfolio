# Question 6:
# Take two numbers and check whether one is a multiple of the other

# Taking input
a = int(input("ENTER FIRST NUMBER: "))
b = int(input("ENTER SECOND NUMBER: "))

# Checking multiple condition
if a % b == 0 or b % a == 0:
    print("ONE IS MULTIPLE OF ANOTHER")
else:
    print("NOT MULTIPLE")
