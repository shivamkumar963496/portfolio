# function define karna
def count(s):
    d = {}

    for ch in s:
        d[ch] = d.get(ch, 0) + 1

    return d

# strings
s1 = "listen"
s2 = "silent"

# compare karna
print(count(s1) == count(s2))
📊 Output:

True
