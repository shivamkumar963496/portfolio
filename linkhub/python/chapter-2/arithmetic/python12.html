# Question 12:
# Principal, rate aur time input lo aur simple interest calculate karo

# user se values input le rahe hain
p = float(input("ENTER PRINCIPAL = "))
r = float(input("ENTER RATE = "))
t = float(input("ENTER TIME = "))

# simple interest calculate kar rahe hain
si = (p * r * t) / 100

# interest aur total amount print kar rahe hain
print("SIMPLE INTEREST =", si)
print("TOTAL AMOUNT WITH INTEREST =", p + si)
