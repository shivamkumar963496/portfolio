# Question 2:
# Write a Python program to take a number as input
# and find the sum of numbers from 1 to n.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# sum variable ko 0 se initialize kiya
total = 0

# loop 1 se n tak chalega aur sum add karega
for i in range(1, n+1):
    total += i

# final sum print kiya
print("SUM =", total)
