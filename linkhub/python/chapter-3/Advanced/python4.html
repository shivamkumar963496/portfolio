# Question 4:
# Check whether both numbers are positive, both negative, or mixed

# Taking first number input
a = int(input("ENTER 1ST NUMBER = "))

# Taking second number input
b = int(input("ENTER 2ND NUMBER = "))

# Checking conditions
if a > 0 and b > 0:
    print("BOTH POSITIVE")
elif a < 0 and b < 0:
    print("BOTH NEGATIVE")
else:
    print("MIXED")
