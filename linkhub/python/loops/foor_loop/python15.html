# Question 15:
# Write a Python program to take a number as input
# and find the largest digit in the number.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# maximum digit store karne ke liye variable
max_digit = 0

# number ke har digit ko check kiya
while n > 0:
    digit = n % 10
    if digit > max_digit:
        max_digit = digit
    n //= 10

# final result print kiya
print("LARGEST DIGIT IN GIVEN NUMBER =", max_digit)
