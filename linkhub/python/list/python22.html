# Question 23:
# Sort list without using sort() (Bubble Sort).

matrix = [1,5,8,7,9,6,5]

for i in range(len(matrix)):
    for j in range(len(matrix)-1):
        if matrix[j] > matrix[j+1]:
            matrix[j], matrix[j+1] = matrix[j+1], matrix[j]

print(matrix)
📊 Output:

[1, 5, 5, 6, 7, 8, 9]
