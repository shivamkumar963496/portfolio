# Do numbers input lo aur remainder formula se calculate karke print karo

# user se numbers input le rahe hain
a = int(input("ENTER FIRST NUMBER = "))

# formula se remainder calculate kar rahe hain

# result print kar rahe hain
print("REMAINDER =", rem)
