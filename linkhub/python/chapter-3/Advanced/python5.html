# Question 5:
# Classify a number into ranges: 0–50, 51–100, 101–200, Above 200

# Taking number input
num = int(input("ENTER A NUMBER = "))

# Checking range
if num <= 50:
    print("0-50")
elif num <= 100:
    print("51-100")
elif num <= 200:
    print("101-200")
else:
    print("ABOVE 200")
