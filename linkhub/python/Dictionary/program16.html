# Question 16:
# String ke har character ka frequency count karo

# string create karna
s = "my name is shivam kumar i love python"

# empty dictionary banana
freq = {}

# loop ke through frequency count karna
for ch in s:
    freq[ch] = freq.get(ch, 0) + 1

# output print karna
print(freq)
