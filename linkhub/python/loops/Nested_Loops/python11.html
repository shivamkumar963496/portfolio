# Question 11:
# Print prime numbers from 2 to 20 using for-else loop.

# loop from 2 to 20
for num in range(2,21):

    # check divisibility
    for i in range(2, num):

        # if divisible then not prime
        if num % i == 0:
            break

    # else runs only if loop not broken
    else:
        print(num)
