# Age input lo aur check karo person youth category me aata hai ya nahi (18 se 25)

# user se age input le rahe hain
age = int(input("Enter age: "))

# youth range condition check kar rahe hain
if age >= 18 and age <= 25:
    print("Youth")
else:
    print("Not Youth")
