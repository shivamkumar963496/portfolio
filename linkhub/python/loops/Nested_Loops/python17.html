# Question 17:
# Print inner stars by skipping border positions using continue.

# outer loop
for i in range(5):

    # inner loop
    for j in range(5):

        # skip border (first & last row/column)
        if i == 0 or i == 4 or j == 0 or j == 4:
            continue

        print("*", end=" ")

    # move to next line
    print()

* * *
