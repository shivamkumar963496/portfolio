# Question 13:
# Write a Python program to check whether a number is prime or not.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# prime check karne ke liye flag variable
flag = True

# 1 ya usse chhote numbers prime nahi hote
if n <= 1:
    flag = False
else:
    # 2 se n-1 tak check kiya
    for i in range(2, n):
        if n % i == 0:
            flag = False
            break

# result print kiya
if flag:
    print("PRIME NUMBER")
else:
    print("NOT PRIME")
