# Question 4:
# Write a Python program to take a number as input
# and count total even numbers from 1 to n.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# even numbers count karne ke liye variable
count = 0

# loop 1 se n tak chalega
for i in range(1, n+1):
    if i % 2 == 0:
        count += 1

# final count print kiya
print("Even count =", count)
