# Question 29:
# Simulate ATM withdrawal with balance check.

# initial balance
balance = 10000

# loop for withdrawal
while True:
    w = int(input("Enter withdrawal amount: "))
    
    if w > balance:
        print("Insufficient balance")
        break   # stop if balance low
    
    balance -= w
    print("Remaining balance:", balance)
