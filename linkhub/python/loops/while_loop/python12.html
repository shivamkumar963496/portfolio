# Question 12:
# Check if a number is Armstrong number.

# input lena
n = int(input("Enter a number: "))

# original value save
temp = n
sum_of_digits_cube = 0

# cube sum calculate
while n != 0:
    last_digit = n % 10
    sum_of_digits_cube += (last_digit ** 3)
    n //= 10

# compare
if temp == sum_of_digits_cube:
    print("Armstrong number")
else:
    print("Not an Armstrong number")
