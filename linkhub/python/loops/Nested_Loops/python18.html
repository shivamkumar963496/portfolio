# Question 18:
# Password check with 3 attempts using break and continue.

# correct password
correct = "1234"

# loop for 3 attempts
for attempts in range(3):

    password = input("ENTER YOUR PASSWORD : ")

    # check password
    if password == correct:
        print("LOGIN SUCCESS")
        break

    else:
        print("WRONG PASSWORD")
        continue
