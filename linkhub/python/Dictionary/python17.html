# Question 17:
# Ek Python program likho jo words ko unke first letter ke basis par group kare.

# words list create karna
words = ["apple", "ant", "bat", "ball"]

# empty dictionary banana
dic = {}

# grouping karna
for w in words:
    key = w[0]

    if key not in dic:
        dic[key] = []

    dic[key].append(w)

# output print karna
print(dic)
📊 Output:

{'a': ['apple', 'ant'], 'b': ['bat', 'ball']}
