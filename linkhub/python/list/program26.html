# Question 26:
# List me se duplicate elements hata kar unique list banani hai

# List create ki
nums = [10,25,5,40,12,36,98,75,45,68,92,91,5,25,10,98]

# Empty list banayi unique elements ke liye
unique_list = []

# Loop chala kar unique elements add kiye
for i in nums:
    if i not in unique_list:   # agar pehle se exist nahi karta
        unique_list.append(i) # to add kar do

# Output print kiya
print(unique_list)
📊 Output:

[10, 25, 5, 40, 12, 36, 98, 75, 45, 68, 92, 91]
