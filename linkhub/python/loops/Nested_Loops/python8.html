# Question 8:
# Search an element in 2D list using nested loop and break.

# 2D list (grid)
grid = [
    [1,2,3],
    [4,5,6],
    [7,8,9]
]

# target element
target = 5

# flag variable
found = False

# outer loop for rows
for row in grid:

    # inner loop for elements
    for element in row:

        # check if element found
        if element == target:
            print("ELEMENT FOUND:", element)
            found = True
            break   # inner loop break
    
    # break outer loop if found
    if found:
        break
📊 Output:

ELEMENT FOUND: 5
