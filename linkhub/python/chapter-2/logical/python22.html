# Temperature input lo aur check karo safe range me hai ya nahi (20 se 30)

# user se temperature input le rahe hain
temp = int(input("Enter temp: "))

# safe range condition check kar rahe hain
if temp >= 20 and temp <= 30:
    print("Safe")
else:
    print("Unsafe")
