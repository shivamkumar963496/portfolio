# Question 15:
# Take two strings from the user and check whether they are equal (case-insensitive)

# Taking string input
s1 = input("ENTER FIRST STRING: ")
s2 = input("ENTER SECOND STRING: ")

# Comparing strings
if s1.lower() == s2.lower():
    print("STRINGS ARE EQUAL")
else:
    print("STRINGS ARE NOT EQUAL")
