# Question 2:
# Check special ticket discount eligibility
# Discount is available if age <= 12 OR age >= 60 OR gender is Female

# Taking age input
age = int(input("ENTER YOUR AGE = "))

# Taking gender input
gender = input("ENTER YOUR GENDER (M/F) = ").lower()

# Checking discount eligibility
if age <= 12 or age >= 60 or gender == "f":
    print("SPECIAL DISCOUNT AVAILABLE")
else:
    print("NO DISCOUNT AVAILABLE")
