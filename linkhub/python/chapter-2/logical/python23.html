# Salary input lo aur check karo mid salary range me hai ya nahi (40000 se 100000)

# user se salary input le rahe hain
salary = int(input("Enter salary: "))

# mid salary range condition check kar rahe hain
if salary > 40000 and not salary > 100000:
    print("mid salary")
else:
    print("out of range")
