# Question 5:
# Take income from the user and calculate tax based on given slabs

# Taking income input
income = int(input("ENTER YOUR INCOME = "))

# Calculating tax
if income < 250000:
    print("NO TAX")
elif income <= 500000:
    print("TAX =", income * 0.05)
elif income <= 1000000:
    print("TAX =", income * 0.10)
else:
    print("TAX =", income * 0.20)
