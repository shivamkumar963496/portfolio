# Question 36:
# List ke har element ki frequency (count) nikalni hai

# List create ki
nums = [10,25,5,40,12,36,98,75,45,68,92,91,5,25,10,98]

# Empty dictionary banayi
freq = {}

# Loop chala kar frequency count ki
for i in nums:
    freq[i] = nums.count(i)   # har element ka count

# Output print kiya
print(freq)
📊 Output:

{10: 2, 25: 2, 5: 2, 40: 1, 12: 1, 36: 1, 98: 2, 75: 1, 45: 1, 68: 1, 92: 1, 91: 1}
