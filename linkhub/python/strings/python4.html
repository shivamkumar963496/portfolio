# Question 4:
# Count number of consonants in a string.

# input string
name = "shivam kumar"

# initialize counter
count = 0

# loop through string
for ch in name:
    if ch.isalpha() and ch not in "aeiouAEIOU":
        count += 1

# print result
print("consonent in string :", count)
📊 Output:

consonent in string : 10
