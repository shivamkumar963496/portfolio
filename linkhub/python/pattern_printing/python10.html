# Question 10:
# Print diamond star pattern using nested loop.

# upper part
for i in range(1,5):

    # space loop
    for j in range(4 - i):
        print(" ", end="")

    # star loop
    for k in range(i):
        print("*", end=" ")

    print()

# lower part
for i in range(3, 0, -1):

    # space loop
    for j in range(4 - i):
        print(" ", end="")

    # star loop
    for k in range(i):
        print("*", end=" ")

    print()
📊 Output:

*
* *
* * *
* * * *
* * *
* *
*
