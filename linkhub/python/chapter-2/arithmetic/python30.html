# 3 digit number input lo aur uska reverse print karo

# user se number input le rahe hain
num = int(input("ENTER A NUMBER = "))

# reverse calculate kar rahe hain
rev = (num % 10) * 100 + ((num // 10) % 10) * 10 + (num // 100)

# result print kar rahe hain
print("REVERSE NUMBER =", rev)
