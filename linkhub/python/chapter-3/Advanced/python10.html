# Question 10:
# Classify performance based on score
# >=90 → Excellent, >=70 → Good, >=50 → Average, <50 → Poor

# Taking score input
score = float(input("ENTER YOUR SCORE = "))

# Performance classification
if score >= 90:
    print("EXCELLENT")
elif score >= 70:
    print("GOOD")
elif score >= 50:
    print("AVERAGE")
else:
    print("POOR")
