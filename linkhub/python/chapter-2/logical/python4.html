# Do numbers input lo aur check karo dono positive hain ya nahi

# user se first number input le rahe hain
a = int(input("Enter first number: "))

# user se second number input le rahe hain
b = int(input("Enter second number: "))

# condition check kar rahe hain
if a > 0 and b > 0:
    print("Both numbers are positive")
else:
    print("One or both numbers are negative")
