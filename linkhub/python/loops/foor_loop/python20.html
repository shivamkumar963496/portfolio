# Question 20:
# Write a Python program to print a number triangle pattern.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# outer loop rows ke liye
for i in range(1, n + 1):
    
    # inner loop numbers print karega
    for j in range(1, i + 1):
        print(j, end="")
    
    # next line ke liye
    print()
