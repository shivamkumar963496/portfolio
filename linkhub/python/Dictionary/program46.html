# Question 46:
# Do dictionaries me same key aur same value wale items nikaalo

# dictionaries create karna
d1 = {"a": 1, "b": 2, "c": 3}
d2 = {"b": 2, "c": 4, "d": 5}

# empty dictionary
common = {}

# matching items find karna
for k in d1:
    if k in d2 and d1[k] == d2[k]:
        common[k] = d1[k]

# output print karna
print(common)
📊 Output:

{'b': 2}
