# Question 24:
# Password validation with length check using nested loop.

# correct password
correct = "shivam123"

# 3 attempts
for attempts in range(3):

    password = input("enter password = ")

    # validation loop
    while True:

        # check password length
        if len(password) < 5:
            print("short password")
            break

        # check correct password
        if password == correct:
            print("Login Successful")
            break

        else:
            print("Wrong Password")
            break

    # break outer loop if login success
    if password == correct:
        break
    else:
        continue
