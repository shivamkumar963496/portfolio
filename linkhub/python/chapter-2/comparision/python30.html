# Ek number input lo aur check karo 500 se bada, equal ya chhota

# user se number input le rahe hain
num = int(input("ENTER NUMBER = "))

# condition check kar rahe hain
if num > 500:
    print("NUMBER IS GREATER THAN 500")
elif num == 500:
    print("NUMBER IS EQUAL TO 500")
else:
    print("NUMBER IS LESS THAN 500")
