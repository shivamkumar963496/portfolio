# Question 14:
# Check if a number is prime.

# input lena
n = int(input("Enter a number: "))

# assume prime
i = 2
flag = True

# check 2 se n-1 tak
while i < n:
    if n % i == 0:
        flag = False   # divisible mila
        break
    i += 1

# result print
if flag and n > 1:
    print("Prime number")
else:
    print("Not a prime number")
