# Age input lo aur check karo 18 se 60 ke beech hai ya nahi

# user se age input le rahe hain
age = int(input("ENTER YOUR AGE = "))

# range condition check kar rahe hain
if age > 18 and age < 60:
    print("AGE IS BETWEEN 18 AND 60")
else:
    print("AGE NOT IN RANGE")
