# Question 37:
# Do dictionaries ko merge karke unki values ka sum nikalo

# dictionaries create karna
d1 = {"a": 1, "b": 2}
d2 = {"b": 3, "c": 4}

# result dictionary
result = {}

# union of keys par loop
for k in set(d1) | set(d2):
    result[k] = d1.get(k, 0) + d2.get(k, 0)

# output print karna
print(result)
📊 Output:

{'a': 1, 'b': 5, 'c': 4}
