# Question 14:
# Take a number from the user and check whether it is Even/Odd and Positive/Negative

# Taking number input
num = int(input("ENTER A NUMBER = "))

# Checking number type
if num > 0:
    if num % 2 == 0:
        print("EVEN AND POSITIVE NUMBER")
    else:
        print("ODD AND POSITIVE NUMBER")
elif num < 0:
    if num % 2 == 0:
        print("EVEN AND NEGATIVE NUMBER")
    else:
        print("ODD AND NEGATIVE NUMBER")
else:
    print("NUMBER IS ZERO")
