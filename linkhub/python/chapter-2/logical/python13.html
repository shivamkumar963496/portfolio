# Ek number input lo aur check karo even hai aur 10 se bada hai ya nahi

# user se number input le rahe hain
num = int(input("Enter number: "))

# condition check kar rahe hain
if num % 2 == 0 and num > 10:
    print("NUMBER IS EVEN AND GREATER THAN 10")
else:
    print("CONDITION NOT MATCH")
