# Question 14:
# Write a Python program to take a number as input
# and print all its factors.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# loop 1 se n tak chalega aur factors check karega
for i in range(1, n+1):
    if n % i == 0:
        print(i, end=" ")
