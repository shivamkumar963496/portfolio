# Question 29:
# Simple ATM system using while loop, break and continue.

# initial balance
balance = 1000

# infinite loop
while True:

    print("\n1. Check Balance")
    print("2. Deposit")
    print("3. Withdraw")
    print("4. Exit")

    choice = int(input("Enter choice: "))

    # exit condition
    if choice == 4:
        print("Thank You")
        break

    # invalid choice
    if choice not in [1,2,3,4]:
        print("Invalid Choice")
        continue

    # check balance
    if choice == 1:
        print("Balance :", balance)

    # deposit
    elif choice == 2:
        amount = int(input("ENTER AMOUNT : "))
        balance += amount
        print("Updated Balance :", balance)

    # withdraw
    elif choice == 3:
        amount = int(input("Enter withdraw: "))

        if amount > balance:
            print("Insufficient Balance")
            continue

        balance -= amount
        print("Updated Balance =", balance)
