# Ek number input lo aur check karo 10 aur 20 ke beech nahi hai (not operator use karke)

# user se number input le rahe hain
num = int(input("Enter number: "))

# range condition check kar rahe hain
if not (num > 10 and num < 20):
    print("Not between 10 and 20")
else:
    print("Between 10 and 20")
