# Question 17:
# Write a Python program to take a string as input
# and count uppercase and lowercase letters.

# user se string input liya ja raha hai
s = input("ENTER A STRING : ")

# uppercase aur lowercase count ke liye variables
upper = 0
lower = 0

# har character ko check kiya
for ch in s:
    if 'A' <= ch <= 'Z':
        upper += 1
    elif 'a' <= ch <= 'z':
        lower += 1

# result print kiya
print("LOWER :", lower)
print("UPPER :", upper)
