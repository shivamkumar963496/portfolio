# Question 14:
# Ek Python program likho jo dictionary me maximum value aur uski key find kare.

# dictionary create karna
d = {"a": 10, "b": 25, "c": 5}

# initial values set karna
max_key = None
max_value = float("-inf")

# loop se max find karna
for k, v in d.items():
    if v > max_value:
        max_value = v
        max_key = k

# output print karna
print(max_key, ":", max_value)
📊 Output:

b : 25
