# Final Project: Advanced ATM System with PIN & History

# correct pin
correct_pin = "1234"

# initial balance
balance = 1000

# transaction history
history = []

# PIN attempts
for attempt in range(3):
    pin = input("ENTER PIN: ")

    if pin == correct_pin:
        print("Login Successful")
        break
    else:
        print("Wrong PIN")

else:
    print("Card Blocked")
    exit()

# ATM Menu
while True:

    print("\n1. Check Balance")
    print("2. Deposit")
    print("3. Withdraw")
    print("4. History")
    print("5. Exit")

    choice = int(input("Enter choice: "))

    if choice == 5:
        print("Thank You")
        break

    if choice == 1:
        print("Balance :", balance)

    elif choice == 2:
        amount = int(input("Enter Deposit Amount: "))
        balance += amount
        history.append(f"Deposited {amount}")
        print("Updated Balance:", balance)

    elif choice == 3:
        amount = int(input("Enter Withdraw Amount: "))

        if amount > balance:
            print("Insufficient Balance")
            continue

        balance -= amount
        history.append(f"Withdraw {amount}")
        print("Updated Balance:", balance)

    elif choice == 4:
        print("Transaction History:")
        for h in history:
            print("-", h)

    else:
        print("Invalid Choice")
