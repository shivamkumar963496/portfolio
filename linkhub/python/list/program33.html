# Question 33:
# List me second maximum element find karna hai (without sorting)

# List create ki
nums = [10,25,5,40,12,36,98,75,45,68,92,91,5,25,10,98]

# Do variables banaye (max aur second max ke liye)
m1 = m2 = float('-inf')

# Loop chala kar max aur second max find kiya
for i in nums:
    if i > m1:
        m2 = m1     # purana max second max ban gaya
        m1 = i      # naya max update hua
        
    elif i > m2 and i != m1:
        m2 = i      # second max update hua

# Output print kiya
print(m2)
