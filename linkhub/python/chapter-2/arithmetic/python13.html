# Question 13:
# 5 subjects ke marks input lo aur total + percentage print karo

# user se subject marks input le rahe hain
hindi = float(input("ENTER HINDI MARKS = "))
english = float(input("ENTER ENGLISH MARKS = "))
mathematics = float(input("ENTER MATHEMATICS MARKS = "))
physics = float(input("ENTER PHYSICS MARKS = "))
chemistry = float(input("ENTER CHEMISTRY MARKS = "))

# total aur percentage calculate kar rahe hain
total_marks = hindi + english + mathematics + physics + chemistry
percentage = total_marks / 5

# result print kar rahe hain
print("TOTAL MARKS =", total_marks)
print("PERCENTAGE =", percentage)
