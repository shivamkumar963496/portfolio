# Question 48:
# 2 se 19 tak prime numbers ko dictionary me store karo

# empty dictionary
d = {}

# prime check karke store karna
for num in range(2, 20):
    is_prime = True

    for i in range(2, num):
        if num % i == 0:
            is_prime = False
            break

    if is_prime:
        d[num] = "prime"

# output print karna
print(d)
📊 Output:

{2: 'prime', 3: 'prime', 5: 'prime', 7: 'prime', 11: 'prime', 13: 'prime', 17: 'prime', 19: 'prime'}
