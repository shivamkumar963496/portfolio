# Question 21:
# Find all pairs with given sum.

nums = [1,2,3,4,5]
target = 6

result = []

for i in range(len(nums)):
    for j in range(i+1, len(nums)):
        if nums[i] + nums[j] == target:
            result.append((nums[i], nums[j]))

print(result)
📊 Output:

[(1, 5), (2, 4)]
