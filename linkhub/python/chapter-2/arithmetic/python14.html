# Salary input lo aur 10% increase ke baad new salary print karo

# user se salary input le rahe hain
salary = float(input("ENTER SALARY = "))

# 10% increase calculate kar rahe hain
increase_salary = salary * 10 / 100
new_salary = salary + increase_salary

# result print kar rahe hain
print("INCREASE SALARY =", increase_salary)
print("NEW SALARY =", new_salary)
