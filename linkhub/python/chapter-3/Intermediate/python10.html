# Question 10:
# Take salary and experience from the user and calculate bonus based on experience

# Taking input
salary = float(input("ENTER YOUR SALARY = "))
experience = int(input("ENTER YOUR EXPERIENCE = "))

# Calculating bonus
if experience <= 2:
    bonus = salary * 0.05
elif experience <= 5:
    bonus = salary * 0.10
else:
    bonus = salary * 0.15

# Printing bonus
print("BONUS =", bonus)
