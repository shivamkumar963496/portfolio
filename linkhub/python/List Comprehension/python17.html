# Question 17:
# List comprehension ka use karke
# string me se numbers aur letters alag karo.

# String create karna
s = "sjkfsjfdklj2325478567uhgjkfdhtjh456yheurhferjthrkhy"

# Digits extract karna
number_result = [ch for ch in s if ch.isdigit()]

# Letters extract karna
letter_result = [ch for ch in s if ch.isalpha()]

# Final output print karna
print(number_result)
print()
print(letter_result)
