# Question 15:
# Classify a number as: Less than 0, Exactly 100, Between 0 and 100, or Greater than 100

# Taking number input
n = int(input("ENTER NUMBER = "))

# Checking conditions
if n < 0:
    print("LESS THAN 0")
elif n == 100:
    print("EXACTLY 100")
elif 0 <= n < 100:
    print("BETWEEN 0 AND 100")
else:
    print("GREATER THAN 100")
