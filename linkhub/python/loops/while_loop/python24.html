# Question 24:
# Print palindromic number pattern.

# input lena
n = int(input("Enter a number: "))

# outer loop (rows)
i = 1

while i <= n:
    j = 1
    
    # increasing part
    while j <= i:
        print(j, end=" ")
        j += 1
    
    # decreasing part
    j = i - 1
    while j >= 1:
        print(j, end=" ")
        j -= 1
        
    print()   # next line
    i += 1
