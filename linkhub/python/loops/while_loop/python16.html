# Question 16:
# Find the maximum digit in a number.

# input lena
n = int(input("Enter a number: "))

# max digit store
max_digit = 0

# digits check
while n != 0:
    digit = n % 10
    if digit > max_digit:
        max_digit = digit
    n //= 10

# result print
print("Maximum digit =", max_digit)
