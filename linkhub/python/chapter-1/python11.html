# Question 11:
# Principal, rate aur time input lo aur simple interest calculate karo

# user se principal amount input le rahe hain
p = float(input("ENTER PRINCIPAL AMOUNT = "))

# user se rate of interest input le rahe hain
r = float(input("ENTER RATE OF INTEREST = "))

# user se time (months me) input le rahe hain
t = float(input("ENTER TIME IN MONTHS = "))

# simple interest calculate kar rahe hain
si = (p * r * t) / 100

# interest aur total amount print kar rahe hain
print("TOTAL INTEREST =", si)
print("THE TOTAL AMOUNT WITH INTEREST =", si + p)
