# Question 6:
# Check whether a character is Uppercase, Lowercase, Digit, or Special Character

# Taking character input
ch = input("ENTER A CHARACTER = ")

# Checking character type
if ch.isupper():
    print("UPPERCASE")
elif ch.islower():
    print("LOWERCASE")
elif ch.isdigit():
    print("DIGIT")
else:
    print("SPECIAL CHARACTER")
