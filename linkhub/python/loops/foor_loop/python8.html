# Question 8:
# Write a Python program to take a number as input
# and find the sum of its digits.

# user se number input liya ja raha hai
num = int(input("ENTER A NUMBER : "))

# sum store karne ke liye variable
total = 0

# number ko string me convert karke har digit ko add kiya
for i in str(num):
    total += int(i)

# final sum print kiya
print("SUM OF DIGITS :", total)
