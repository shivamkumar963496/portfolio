# Question 22:
# Dictionary me unique values ke basis par filtering karo

# dictionary create karna
d = {"a": 1, "b": 2, "c": 1}

# empty dictionary banana
unique = {}

# unique values filter karna
for k, v in d.items():
    if v not in unique.values():
        unique[k] = v

# output print karna
print(unique)
📊 Output:

{'a': 1, 'b': 2}
