# Question 22:
# Print numbers with Fizz-Buzz logic in matrix form.

# starting number
num = 1

# outer loop
for i in range(5):

    # inner loop
    for j in range(5):

        # check fizz-buzz (divisible by 3 and 5)
        if num % 3 == 0 and num % 5 == 0:
            print("fizz-buzz", end=" ")
            num += 1
            continue

        # check fizz (divisible by 3)
        if num % 3 == 0:
            print("fizz", end=" ")
            num += 1
            continue

        # check buzz (divisible by 5)
        if num % 5 == 0:
            print("buzz", end=" ")
            num += 1
            continue

        # print number
        print(num, end=" ")
        num += 1

    # move to next line
    print()
