# Question 19:
# Take speed from the user and classify it as SLOW, NORMAL, or FAST

# Taking speed input
speed = int(input("ENTER SPEED = "))

# Checking speed category
if speed < 30:
    print("SLOW SPEED")
elif speed <= 70:
    print("NORMAL SPEED")
else:
    print("FAST SPEED")
