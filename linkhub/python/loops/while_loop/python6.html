# Question 6:
# Print table of a number.

# input lena
n = int(input("ENTER A NUMBER = "))

# start 1 se
i = 1

# 1 se 10 tak loop
while i <= 10:
    print(n, "X", i, "=", n * i)   # table print
    i += 1   # next step
