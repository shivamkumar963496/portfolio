# Question 35:
# String ke words ka frequency count dictionary me store karo

# string create karna
text = "python code python is easy code"

# words me convert karna
words = text.split()

# empty dictionary
freq = {}

# frequency count karna
for w in words:
    freq[w] = freq.get(w, 0) + 1

# output print karna
print(freq)
📊 Output:

{'python': 2, 'code': 2, 'is': 1, 'easy': 1}
