# Question 45:
# String me uppercase aur lowercase characters count karo

# string create karna
s = "My name Is Shivam Kumar And I Am Learning Python"

# dictionary create karna
d = {"upper": 0, "lower": 0}

# count karna
for ch in s:
    if ch.isupper():
        d["upper"] += 1
    elif ch.islower():
        d["lower"] += 1

# output print karna
print(d)
📊 Output:

{'upper': 7, 'lower': 39}
