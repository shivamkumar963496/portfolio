# Question 7:
# Write a Python program to take a number as input
# and count total digits in the number.

# user se number input liya ja raha hai
n = int(input("ENTER A NUMBER : "))

# digit count karne ke liye variable
count = 0

# number ko string me convert karke har digit count kiya
for i in str(n):
    count += 1

# total digits print kiya
print("TOTAL DIGITS :", count)
