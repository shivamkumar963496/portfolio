# Question 13:
# Print square star pattern of size n, stop if n == 0.

# input size
n = int(input("Enter size: "))

# check if n is 0
if n == 0:
    print("Program Stopped")

else:
    # outer loop for rows
    for i in range(n):

        # inner loop for columns
        for j in range(n):
            print("*", end=" ")

        # move to next line
        print()
📊 Output:

Enter size: 4
* * * *
* * * *
* * * *
* * * *
