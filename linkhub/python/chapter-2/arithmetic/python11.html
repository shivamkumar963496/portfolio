# Circle ka radius input lo aur area print karo

# user se radius input le rahe hain
r = float(input("ENTER THE RADIUS OF CIRCLE = "))

# area calculate kar rahe hain (pi = 3.1415)

# result print kar rahe hain
print("THE AREA OF THE CIRCLE =", area)
