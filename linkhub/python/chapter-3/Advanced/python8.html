# Question 8:
# Check whether a number is an Armstrong number (for any number of digits)

# Taking number input
n = int(input("ENTER A NUMBER = "))

# Calculating number of digits
digits = len(str(n))

# Calculating Armstrong sum
temp = n
armstrong_sum = 0

while temp > 0:
    digit = temp % 10
    armstrong_sum += digit ** digits
    temp //= 10

# Checking Armstrong number
if armstrong_sum == n:
    print("ARMSTRONG NUMBER")
else:
    print("NOT ARMSTRONG NUMBER")
